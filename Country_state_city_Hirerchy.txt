-- solution

with city_level as
(
select distinct t2.Country, t2.State, t2.City, t1.Sales from input_table_2 t2
inner join input_table_1 t1
on t2.city = t1.Market
), city_level_state_sales as 
(
select state, sum(sales) as sales
from city_level
group by state
)
, state_level
as
(
select distinct t2.Country, t2.State, isnull(t1.Sales,0) as sales
from input_table_2 t2 left join
input_table_1 t1 on t2.State = t1.Market
), state_extra_sales as
(
select ss.Country, ss.State, ss.sales - isnull(css.sales,0) as sales 
from state_level ss left join 
city_level_state_sales css on
ss.State = css.State
where ss.sales != isnull(css.sales,0)
), state_and_city_sales as 
(
select country, State, null as city,sales from state_extra_sales
union all
select * from city_level
)
select * from state_and_city_sales
union all
select a.Country, null as state, null as city, a.Sales - b.sales as sales from 
(
select distinct t2.Country, t1.Sales from input_table_2 t2 inner join 
input_table_1 t1 on t2.Country = t1.Market) a
inner join (select country, sum(sales) as sales from state_and_city_sales group by country) b
on a.Country = b.Country
